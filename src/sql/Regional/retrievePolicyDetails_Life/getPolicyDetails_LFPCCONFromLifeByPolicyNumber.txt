SELECT
	CCON.POLICY_CONVERSION_NUMBER,
	
	CCON.Q3OPNO,
	CCON.Q3TEFF
FROM
	(
	SELECT
		Q3PNO AS POLICY_CONVERSION_NUMBER,
		
		Q3OPNO,
		Q3TEFF,
		ROW_NUMBER() OVER (PARTITION BY Q3OPNO ORDER BY Q3TEFF DESC) AS REC
	FROM MR_LFPCCON
	WHERE Q3OPNO= '%policyNumber%' /*INPUT POLICY NUMBER*/
	)CCON
WHERE CCON.REC=1
