SELECT 
X5PNO AS POLICY_NUMBER,
X5SPCODE AS SPECIAL_CASE_CODE,
X5SPTYPE AS SPECIAL_CASE_TYPE,
DECODE(IS_DATE_DAPI(TO_CHAR(X5UDATE)),1,TO_DATE(TO_CHAR(X5UDATE),'YYYY-MM-DD'),TO_DATE('19000101','YYYY-MM-DD')) AS UPDATED_DATE,
X5UUSER AS UPDATED_USER,

X5BNK,
X5RMK
FROM MR_LFPBNKUP
WHERE X5PNO=? /*INPUT POLICY NUMBER*/
ORDER BY UPDATED_DATE ASC
