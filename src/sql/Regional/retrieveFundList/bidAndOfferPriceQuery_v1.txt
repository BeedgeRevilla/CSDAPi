SELECT 
  FMP.FPBIDP,
  FMP.FPOFFP
FROM
  (
  SELECT
    FUNDID,
    FPEFF,
    FPEXP,
    FPDEAL,
    FPOFFP,
    FPISSP,
    FPBIDP,
    FPVAL,
    ROW_NUMBER() OVER (PARTITION BY FUNDID ORDER BY FPEFF DESC) AS RN
  FROM MR_FMPPRCH
  ) FMP
INNER JOIN MR_LFPPMFN LFP
ON TRIM(LFP.M4CODE)=TRIM(FMP.FUNDID)
WHERE FMP.RN=1 AND LFP.M4PNO='%policyNumber%'
