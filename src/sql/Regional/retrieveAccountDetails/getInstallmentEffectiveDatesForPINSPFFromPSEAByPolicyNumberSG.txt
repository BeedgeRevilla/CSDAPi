SELECT 
INSTFROM, /*ADDED FOR AUTOMATION*/
DECODE(IS_DATE(TO_CHAR(INSTFROM)),1,TO_DATE(TO_CHAR(INSTFROM),'YYYY-MM-DD'),TO_DATE('19000101','YYYY-MM-DD')) AS INSTFROM 
FROM 
    (SELECT CHDRNUM, CHDRCOY, INSTFROM, MAX(TRANNO)
    FROM MR_PINSPF 
    WHERE CHDRNUM='%policyNumber%' AND VALIDFLAG='1'
    GROUP BY CHDRNUM,CHDRCOY,INSTFROM
    ORDER BY INSTFROM
    )
