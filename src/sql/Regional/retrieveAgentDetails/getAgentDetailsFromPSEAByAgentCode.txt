SELECT
AGNT.AGNTNUM,
DECODE(IS_DATE(TO_CHAR(AGNT.START_DATE)),1,TO_DATE(TO_CHAR(AGNT.START_DATE),'YYYY-MM-DD'),TO_DATE('19000101','YYYY-MM-DD')) AS CONTRACT_EFFECTIVE_DATE,
AGNT.AGNTBR,
AGNT.AGTYPE,
AGNT.CHDRSTCDA,
AGPL.AGTLICNO,
AGNT.ZDISTRICT,
AGNT.CLNTNUM,
DES.LONGDESC,
DECODE(IS_DATE(TO_CHAR(AGNT.DATEEND)),1,TO_DATE(TO_CHAR(AGNT.DATEEND),'YYYY-MM-DD'),TO_DATE('19000101','YYYY-MM-DD')) AS TERMINATION_DATE 

FROM
MR_AGNTPF AGNT LEFT OUTER JOIN MR_AGPLPF AGPL
ON AGNT.AGNTNUM=AGPL.AGNTNUM
LEFT OUTER JOIN
(SELECT * FROM MR_DESCPF DES WHERE DES.DESCTABL='TR889') DES
ON DES.DESCITEM = AGNT.ZDISTRICT
WHERE AGNT.AGNTNUM = '%agentCode%' /*VALUE*/
AND AGNT.AGNTCOY='1'
/*${companyCodeString}*/

AND AGNT.VALIDFLAG='1'
